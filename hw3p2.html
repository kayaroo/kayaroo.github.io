JavaScript Logic (script.js): function calculateEarnings() { // Get input values const salespersonName = document.getElementById("salespersonName").value; const item1Qty = parseFloat(document.getElementById("item1Qty").value); const item2Qty = parseFloat(document.getElementById("item2Qty").value); const item3Qty = parseFloat(document.getElementById("item3Qty").value); const item4Qty = parseFloat(document.getElementById("item4Qty").value); // Item prices const itemPrices = [20.99, 12.75, 9.95, 35.89]; // Calculate total sales const totalSales = (item1Qty * itemPrices[0]) + (item2Qty * itemPrices[1]) + (item3Qty * itemPrices[2]) + (item4Qty * itemPrices[3]); // Calculate commission const commission = 0.09 * totalSales; // Calculate total earnings const totalEarnings = 250 + commission; // Display results (formatted to 2 decimal places) const resultsDiv = document.getElementById("results"); resultsDiv.innerHTML = `

Results for ${salespersonName}:

Total Sales: $${totalSales.toFixed(2)}

Commission: $${commission.toFixed(2)}

Total Earnings: $${totalEarnings.toFixed(2)}

`; }
